\section{Newton's First Law}
	\begin{align}
		\vec{v} = \text{constant}
		iff
		\vec{F_{net}} = 0
		\vec{F_{net}} = \text{net force}
		= \sum \vec{F} = \text{sum of all forces}
	\end{align}
