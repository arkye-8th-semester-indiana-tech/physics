\section{Problem (6)}
	What is the magnitude of the acceleration of a sprinter running at $27 \ ft/s$ when rounding a turn with a radius of $55 \ ft$?

	\textbf{R:} \newline
	\begin{align}
		a_{c} = \ &\frac{v^{2}}{r}& \notag \\
		a_{c} = \ &\frac{(27 \ ft/s)^{2}}{55 \ ft} = 13.2545 \ ft/s^{2}&
	\end{align}
